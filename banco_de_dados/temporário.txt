CREATE EXTENSION IF NOT EXISTS postgis;

-- vai ter q mudar né, mas essa base ta boa
CREATE TABLE objects (
    id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,

    -- localização geográfica original
    geom geometry(Point, 4326) NOT NULL,

    -- projeção métrica para grid fixa
    geom_3857 geometry(Point, 3857) NOT NULL,

    -- agrupamento menor (100m x 100m)
    cell_x BIGINT NOT NULL,
    cell_y BIGINT NOT NULL,

    -- grupo maior (1000m x 1000m)
    region_x BIGINT NOT NULL,
    region_y BIGINT NOT NULL,

    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_objects_geog_gist
    ON objects
    USING GIST ((geom::geography));

CREATE INDEX idx_objects_geom3857_gist
    ON objects
    USING GIST (geom_3857);

CREATE INDEX idx_objects_region_xy
    ON objects (region_x, region_y);

CREATE INDEX idx_objects_cell_xy
    ON objects (cell_x, cell_y);

CREATE INDEX idx_objects_region_cell_xy
    ON objects (region_x, region_y, cell_x, cell_y);

-- Revisado - Tudo certo
CREATE OR REPLACE FUNCTION trg_objects_set_grid()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_geom_3857 geometry(Point, 3857);
    v_cell_size_m CONSTANT INTEGER := 100;
    v_region_size_m CONSTANT INTEGER := 1000;
BEGIN
    IF NEW.geom IS NULL THEN
        RAISE EXCEPTION 'geom não pode ser NULL';
    END IF;

    -- Normaliza SRID
    IF ST_SRID(NEW.geom) = 0 THEN
        NEW.geom := ST_SetSRID(NEW.geom, 4326);
    ELSIF ST_SRID(NEW.geom) <> 4326 THEN
        NEW.geom := ST_Transform(NEW.geom, 4326);
    END IF;

    -- Projeta para metros
    v_geom_3857 := ST_Transform(NEW.geom, 3857);

    NEW.geom_3857 := v_geom_3857;

    -- Grid menor: 100m x 100m
    NEW.cell_x := floor(ST_X(v_geom_3857) / v_cell_size_m)::bigint;
    NEW.cell_y := floor(ST_Y(v_geom_3857) / v_cell_size_m)::bigint;

    -- Grid maior: 1000m x 1000m
    NEW.region_x := floor(ST_X(v_geom_3857) / v_region_size_m)::bigint;
    NEW.region_y := floor(ST_Y(v_geom_3857) / v_region_size_m)::bigint;

    RETURN NEW;
END;
$$;

-- Aparentemente está tudo certo
CREATE TRIGGER objects_set_grid_before_ins_upd
BEFORE INSERT OR UPDATE OF geom
ON objects
FOR EACH ROW
EXECUTE FUNCTION trg_objects_set_grid();

CREATE OR REPLACE FUNCTION achar_todos_em_raio(
    p_lat DOUBLE PRECISION,
    p_lon DOUBLE PRECISION,
    p_raio_m DOUBLE PRECISION
)
RETURNS TABLE (
    region_x BIGINT,
    region_y BIGINT,
    cell_x BIGINT,
    cell_y BIGINT,
    object_id BIGINT,
    object_name TEXT,
    latitude DOUBLE PRECISION,
    longitude DOUBLE PRECISION,
    distance_m DOUBLE PRECISION
)
LANGUAGE sql
STABLE
AS $$
WITH search_point AS (
    SELECT
        ST_SetSRID(ST_MakePoint(p_lon, p_lat), 4326) AS geom_4326,
        ST_Transform(ST_SetSRID(ST_MakePoint(p_lon, p_lat), 4326), 3857) AS geom_3857
),
region_bounds AS (
    SELECT
        floor((ST_X(geom_3857) - p_raio_m) / 1000)::bigint AS min_region_x,
        floor((ST_X(geom_3857) + p_raio_m) / 1000)::bigint AS max_region_x,
        floor((ST_Y(geom_3857) - p_raio_m) / 1000)::bigint AS min_region_y,
        floor((ST_Y(geom_3857) + p_raio_m) / 1000)::bigint AS max_region_y,
        geom_4326
    FROM search_point
),
candidate_regions AS (
    SELECT
        rx AS region_x,
        ry AS region_y,
        rb.geom_4326
    FROM region_bounds rb
    CROSS JOIN generate_series(rb.min_region_x, rb.max_region_x) AS rx
    CROSS JOIN generate_series(rb.min_region_y, rb.max_region_y) AS ry
),
-- Transformar para candidate_cell, por enquanto vê os individuos individuais, não no agrupamento de célula
candidate_objects AS (
    SELECT
        o.*,
        cr.geom_4326 AS search_geom
    FROM candidate_regions cr
    JOIN objects o
      ON o.region_x = cr.region_x
     AND o.region_y = cr.region_y
),
filtered_objects AS (
    SELECT
        o.region_x,
        o.region_y,
        o.cell_x,
        o.cell_y,
        o.id AS object_id,
        o.name AS object_name,
        ST_Y(o.geom) AS latitude,
        ST_X(o.geom) AS longitude,
        ST_Distance(o.geom::geography, o.search_geom::geography) AS distance_m
    FROM candidate_objects o
    WHERE ST_DWithin(o.geom::geography, o.search_geom::geography, p_raio_m)
)
SELECT
    region_x,
    region_y,
    cell_x,
    cell_y,
    object_id,
    object_name,
    latitude,
    longitude,
    distance_m
FROM filtered_objects
ORDER BY region_y, region_x, cell_y, cell_x, distance_m;
$$;
